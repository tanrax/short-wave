#+BEGIN_SRC bash
pip install -r requeriments.txt
cp envExample .env
#+END_SRC

Edit the configuration file

#+BEGIN_SRC bash
vi .env
#+END_SRC

** Run

#+BEGIN_SRC bash
python app.py send
